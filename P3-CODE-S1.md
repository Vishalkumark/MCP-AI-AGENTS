```python
"""
thread_client.py
================
Fetches email threads (conversation chains) from Microsoft Graph API.

Why this file exists:
    MOM generation needs the full conversation context — not just one
    email but the entire chain of replies. Graph API groups related
    emails by conversationId. This file fetches all messages in a
    thread using that ID.

Design notes:
    - Each email in Outlook has a conversationId field that links all
      replies in the same thread together.
    - We fetch all messages sharing a conversationId, sorted oldest
      first so the MOM reflects the natural flow of the conversation.
    - Thread fetch is limited to 50 messages per conversation to
      prevent excessively large payloads.
"""

# ---------------------------------------------------------------------------
# Imports
# ---------------------------------------------------------------------------
from graph.graph_client_factory import get_graph_client
from utils.logger import get_logger

logger = get_logger(__name__)


# ---------------------------------------------------------------------------
# Function: get_email_thread
# ---------------------------------------------------------------------------
async def get_email_thread(message_id: str) -> list[dict]:
    """
    Fetch all emails in the same conversation thread as the given email.

    Args:
        message_id (str): Graph API ID of any email in the thread.

    Returns:
        list[dict]: All emails in the thread, sorted oldest first.
                     Each dict contains id, subject, from, to,
                     receivedDateTime, body content, and bodyPreview.
    """
    logger.info(f"Fetching email thread for message: {message_id}")
    client = get_graph_client()

    # Step 1: Get the target email to extract its conversationId
    message = await client.me.messages.by_message_id(message_id).get()

    if not message:
        logger.warning(f"Message {message_id} not found")
        return []

    conversation_id = message.conversation_id

    if not conversation_id:
        logger.info("No conversationId found — returning single email")
        return [_format_message(message)]

    logger.info(f"ConversationId: {conversation_id} — fetching full thread")

    # Step 2: Search for all messages with the same conversationId
    from msgraph.generated.users.item.messages.messages_request_builder import (
        MessagesRequestBuilder,
    )

    query_params = MessagesRequestBuilder.MessagesRequestBuilderGetQueryParameters(
        filter=f"conversationId eq '{conversation_id}'",
        orderby=["receivedDateTime ASC"],
        top=50,
    )
    request_config = MessagesRequestBuilder.MessagesRequestBuilderGetRequestConfiguration(
        query_parameters=query_params,
    )
    request_config.headers.add("ConsistencyLevel", "eventual")

    response = await client.me.messages.get(request_configuration=request_config)
    messages = response.value if response and response.value else []

    logger.info(f"Thread contains {len(messages)} messages")
    return [_format_message(m) for m in messages]


# ---------------------------------------------------------------------------
# Helper: _format_message
# ---------------------------------------------------------------------------
def _format_message(message) -> dict:
    """
    Convert a Graph SDK Message object to a clean dict for MOM processing.
    """
    # Extract TO recipients as a list
    to_recipients = []
    if message.to_recipients:
        for r in message.to_recipients:
            if r.email_address:
                to_recipients.append({
                    "name": r.email_address.name or "",
                    "address": r.email_address.address or "",
                })

    return {
        "id": message.id,
        "subject": message.subject or "",
        "from_name": (
            message.from_.email_address.name
            if message.from_ and message.from_.email_address else ""
        ),
        "from_address": (
            message.from_.email_address.address
            if message.from_ and message.from_.email_address else ""
        ),
        "to_recipients": to_recipients,
        "received_date_time": (
            message.received_date_time.isoformat()
            if message.received_date_time else ""
        ),
        "body_content": message.body.content if message.body else "",
        "body_preview": message.body_preview or "",
        "has_attachments": message.has_attachments or False,
        "conversation_id": message.conversation_id or "",
    }
```

```python
"""
tasks_client.py
===============
Microsoft Graph API calls for Microsoft To-Do and Microsoft Planner.

Why this file exists:
    Phase 3 adds the ability to create tasks from emails — both as
    personal To-Do items and as team Planner tasks. These are two
    separate Graph API surfaces with different endpoints and models,
    so they are contained here rather than scattered across files.

Microsoft To-Do vs Microsoft Planner:
    - To-Do: Personal task list, linked to your account.
             Endpoint: /me/todo/lists/{listId}/tasks
    - Planner: Team task board, linked to a Microsoft 365 Group.
               Endpoint: /planner/tasks

Design notes:
    - To-Do tasks are created in the default task list unless a
      specific list is specified.
    - Planner tasks require a planId and bucketId — both configured
      in .env via PLANNER_PLAN_ID and resolved at runtime.
    - Both services require Tasks.ReadWrite permission.
"""

# ---------------------------------------------------------------------------
# Imports
# ---------------------------------------------------------------------------
from datetime import datetime, timezone

from graph.graph_client_factory import get_graph_client
from config.settings import settings

        list[dict]: List of task lists with id and display name.

# Function: create_todo_task
# ---------------------------------------------------------------------------
async def create_todo_task(
    title: str,
    body_text: str = "",
    due_date: str = "",
    list_id: str = "",
) -> dict:
    """
    Create a task in Microsoft To-Do.

    Args:
        title (str): The task title / action item text.
        body_text (str): Optional additional notes or context.
        due_date (str): Optional due date in ISO format or natural
                         language e.g. "2026-07-20", "end of week".
        list_id (str): To-Do list ID to add the task to. If empty,
                        uses the default Tasks list.

    Returns:
        dict: Created task ID, title, and confirmation.
    """
    logger.info(f"Creating To-Do task: '{title}'")
    client = get_graph_client()

    from msgraph.generated.models.todo_task import TodoTask
    from msgraph.generated.models.item_body import ItemBody
    from msgraph.generated.models.body_type import BodyType
    from msgraph.generated.models.date_time_time_zone import DateTimeTimeZone
    from utils.date_utils import parse_date_string

    task = TodoTask()
    task.title = title

    # Add body/notes if provided
    if body_text:
        task.body = ItemBody()
        task.body.content = body_text
        task.body.content_type = BodyType.Text

    # Add due date if provided
    if due_date:
        parsed_dt = parse_date_string(due_date)
        if parsed_dt:
            task.due_date_time = DateTimeTimeZone()
            task.due_date_time.date_time = parsed_dt.strftime("%Y-%m-%dT%H:%M:%S")
            task.due_date_time.time_zone = "UTC"

    # Resolve which list to use
    if not list_id:
        # Use the default "Tasks" list
        all_lists = await get_todo_lists()
        default_list = next(
            (lst for lst in all_lists
             if lst["name"].lower() in ("tasks", "aufgaben", "tâches")),
            all_lists[0] if all_lists else None
        )
        list_id = default_list["id"] if default_list else None

    if not list_id:
        raise ValueError("No To-Do list found. Please ensure Microsoft To-Do is set up.")

    created = await (
        client.me.todo.lists.by_todo_task_list_id(list_id)
        .tasks.post(task)
    )

    return {
        "task_id": created.id,
        "title": created.title,
        "list_id": list_id,
        "due_date": due_date or "Not set",
        "status": "Task created in Microsoft To-Do",
    }

async def get_todo_tasks(list_id: str = "", top: int = 20) -> list[dict]:
    """
    Get tasks from a Microsoft To-Do list.

    Args:
        list_id (str): To-Do list ID. Uses default list if empty.
        top (int): Maximum number of tasks to return.

    Returns:
        list[dict]: List of task dictionaries.
    """
    logger.info("Fetching To-Do tasks")
    client = get_graph_client()

    if not list_id:
        all_lists = await get_todo_lists()
        default_list = next(
            (lst for lst in all_lists
             if lst["name"].lower() in ("tasks", "aufgaben", "tâches")),
            all_lists[0] if all_lists else None
        )
        list_id = default_list["id"] if default_list else None

    if not list_id:
        return []

    from msgraph.generated.users.item.todo.lists.item.tasks.tasks_request_builder import (
        TasksRequestBuilder,
    )

    query_params = TasksRequestBuilder.TasksRequestBuilderGetQueryParameters(
        top=top,
        filter="status ne 'completed'",
        orderby=["createdDateTime DESC"],
    )
    request_config = TasksRequestBuilder.TasksRequestBuilderGetRequestConfiguration(
        query_parameters=query_params
    )

    response = await (
        client.me.todo.lists.by_todo_task_list_id(list_id)
        .tasks.get(request_configuration=request_config)
    )

    tasks = response.value if response and response.value else []

    return [
        {
            "id": t.id,
            "title": t.title,
            "status": t.status.value if t.status else "notStarted",
            "due_date": (
                t.due_date_time.date_time[:10]
                if t.due_date_time else "Not set"
            ),
            "created": (
                t.created_date_time.isoformat()
                if t.created_date_time else ""
            ),
        }
        for t in tasks
    ]

# Function: create_planner_task
# ---------------------------------------------------------------------------
async def create_planner_task(
    title: str,
    notes: str = "",
    due_date: str = "",
    assigned_to_email: str = "",
    bucket_name: str = "",
) -> dict:
    """
    Create a task in Microsoft Planner on the configured team plan.

    Args:
        title (str): Task title.
        notes (str): Additional notes or description.
        due_date (str): Due date in ISO format or natural language.
        assigned_to_email (str): Email of the person to assign the task to.
        bucket_name (str): Name of the Planner bucket/column to add to.
                            Uses first available bucket if not specified.

    Returns:
        dict: Created task ID, title, and confirmation.
    """
    logger.info(f"Creating Planner task: '{title}'")
    client = get_graph_client()

    plan_id = settings.PLANNER_PLAN_ID
    if not plan_id:
        raise ValueError(
            "PLANNER_PLAN_ID not set in .env. "
            "Please configure it with your Planner plan ID."
        )

    from msgraph.generated.models.planner_task import PlannerTask
    from msgraph.generated.models.planner_applied_categories import PlannerAppliedCategories
    from msgraph.generated.models.planner_assignments import PlannerAssignments
    from msgraph.generated.models.planner_assignment import PlannerAssignment
    from utils.date_utils import parse_date_string

    task = PlannerTask()
    task.plan_id = plan_id
    task.title = title

    # Set due date if provided
    if due_date:
        parsed_dt = parse_date_string(due_date)
        if parsed_dt:
            task.due_date_time = parsed_dt.replace(
                tzinfo=timezone.utc
            ).isoformat()

    # Resolve bucket ID if bucket name specified
    if bucket_name:
        from msgraph.generated.planner.plans.item.buckets.buckets_request_builder import (
            BucketsRequestBuilder,
        )
        buckets_response = await (
            client.planner.plans.by_planner_plan_id(plan_id)
            .buckets.get()
        )
        buckets = buckets_response.value if buckets_response else []
        matched_bucket = next(
            (b for b in buckets
             if b.name and b.name.lower() == bucket_name.lower()),
            buckets[0] if buckets else None
        )
        if matched_bucket:
            task.bucket_id = matched_bucket.id

    # Assign to user if email provided
    if assigned_to_email:
        # Resolve email to user ID
        user_response = await client.users.get()
        users = user_response.value if user_response else []
        matched_user = next(
            (u for u in users
             if u.mail and u.mail.lower() == assigned_to_email.lower()),
            None
        )
        if matched_user:
            assignments = PlannerAssignments()
            assignment = PlannerAssignment()
            assignment.odata_type = "#microsoft.graph.plannerAssignment"
            assignments.additional_data = {
                matched_user.id: {
                    "@odata.type": "#microsoft.graph.plannerAssignment",
                    "orderHint": " !"
                }
            }
            task.assignments = assignments

    created = await client.planner.tasks.post(task)

    # Add notes as task details if provided
    if notes and created.id:
        from msgraph.generated.models.planner_task_details import PlannerTaskDetails
        details = PlannerTaskDetails()
        details.description = notes
        await (
            client.planner.tasks.by_planner_task_id(created.id)
            .details.patch(details)
        )

    return {
        "task_id": created.id,
        "title": created.title,
        "plan_id": plan_id,
        "due_date": due_date or "Not set",
        "assigned_to": assigned_to_email or "Unassigned",
        "status": "Task created in Microsoft Planner",
    }
```

```python
"""
mom_tools.py
============
MCP tools for generating formal Minutes of Meeting (MOM) from email
threads or single emails containing meeting notes.

Why this file exists:
    Preparing MOM is one of the most time-consuming post-meeting tasks.
    This tool automates it by reading the email thread or meeting notes
    email, extracting structured content, and producing a formal MOM
    document following a consistent professional template.

MOM format (formal):
    1. Meeting Details
    2. Attendees table
    3. Discussion Summary
    4. Decisions Made
    5. Action Items table
    6. Next Steps
    7. Notes

Design notes:
    - MOM generation relies on LibreChat's LLM for content extraction
      and structuring — no Requesty calls.
    - The tool provides the raw email content + a detailed structured
      instruction so LibreChat produces a consistent, formal MOM.
    - Multi-language: instruction asks LibreChat to detect the email
      language and generate the MOM in the same language.
    - save_mom_as_draft saves the MOM to Outlook Drafts so it can
      be shared with attendees directly.
"""

# ---------------------------------------------------------------------------
# Imports
# ---------------------------------------------------------------------------
from datetime import datetime

from tools.mcp_instance import mcp

from graph.thread_client import get_email_thread
from graph.mail_client import fetch_message_by_id
from graph.draft_client import create_draft

from utils.error_handler import format_tool_error

# ---------------------------------------------------------------------------
# MOM formal template structure passed to LibreChat's LLM
# ---------------------------------------------------------------------------
MOM_INSTRUCTION_TEMPLATE = """
Generate a formal Minutes of Meeting (MOM) document from the email
content provided below. Follow these strict formatting rules:

LANGUAGE RULE: Detect the primary language of the email content and
generate the ENTIRE MOM in that same language. If mixed languages,
use the dominant one.

OUTPUT FORMAT — use exactly this structure:

═══════════════════════════════════════════════════
         MINUTES OF MEETING
═══════════════════════════════════════════════════

1. MEETING DETAILS
   | Field       | Value                          |
   |-------------|--------------------------------|
   | Subject     | [from email subject]           |
   | Date        | [from email date]              |
   | Reference   | [email subject line]           |
   | Prepared by | Outlook AI Agent               |
   | Prepared on | {timestamp}                    |

2. ATTENDEES
   | Name | Email | Role |
   |------|-------|------|
   [Extract from From/To/CC fields of all emails in thread]
   [If role unknown, write "Participant"]

3. DISCUSSION SUMMARY
   [Write 3-5 formal sentences summarising the key topics discussed.
    Use third person. Be factual and concise.]

4. DECISIONS MADE

   |---|-------------|-------|----------|--------|

6. NEXT STEPS

───────────────────────────────────────────────────
   Generated by Outlook AI Agent | {timestamp}
───────────────────────────────────────────────────

EMAIL CONTENT TO PROCESS:
{email_content}
"""


# ---------------------------------------------------------------------------
# Tool: generate_mom
# ---------------------------------------------------------------------------
@mcp.tool
async def generate_mom(
    email_id: str,
    use_thread: bool = True,
) -> dict:
    """
    Generate a formal Minutes of Meeting (MOM) document from an email
    or email thread. Supports multi-language — detects and matches
    the language of the email content.

    Use this tool when the user asks things like:
    - "Generate MOM from this email thread"
    - "Prepare minutes of meeting from John's email"
    - "Create MOM from the project review email"
    - "Extract meeting minutes from this conversation"

    Args:
        email_id (str): Graph API ID of the email (or any email in
                         the thread). Obtained from list_emails or
                         search_emails.
        use_thread (bool): If True, fetches the full conversation
                            thread for richer MOM content. If False,
                            uses only the single pointed email.
                            Default True.

    Returns:
        dict with formatted email content and MOM generation instruction
        for LibreChat's LLM to produce the final formal document.
    """
    logger.info(
        f"Tool called: generate_mom "
        f"(email_id={email_id[:20]}..., use_thread={use_thread})"
    )

    try:
        # Step 1: Fetch email content
        if use_thread:
            messages = await get_email_thread(email_id)
            source_label = f"thread ({len(messages)} messages)"
        else:
            single = await fetch_message_by_id(email_id)
            messages = [{
                "subject": single.get("subject", ""),
                "from_name": single.get("from", {}).get("emailAddress", {}).get("name", ""),
                "from_address": single.get("from", {}).get("emailAddress", {}).get("address", ""),
                "to_recipients": [],
                "received_date_time": single.get("receivedDateTime", ""),
                "body_content": single.get("body", {}).get("content", ""),
                "body_preview": single.get("bodyPreview", ""),
            }]
            source_label = "single email"

        if not messages:
            return {
                "error": True,
                "message": "No email content found to generate MOM from.",
            }

        # Step 2: Build a clean structured email content block
        # Strip HTML tags for cleaner LLM processing
        import re

        def strip_html(html: str) -> str:
            text = re.sub(r"<[^>]+>", " ", html or "")
            return re.sub(r"\s+", " ", text).strip()

        email_content_parts = []
        all_participants = set()

        for i, msg in enumerate(messages, 1):
            subject = msg.get("subject", "")
            from_name = msg.get("from_name", "")
            from_addr = msg.get("from_address", "")
            received = msg.get("received_date_time", "")[:10]
            body = strip_html(msg.get("body_content", "") or msg.get("body_preview", ""))

            if from_addr:
                all_participants.add(f"{from_name} <{from_addr}>")

            for r in msg.get("to_recipients", []):
                if r.get("address"):
                    all_participants.add(
                        f"{r.get('name', '')} <{r.get('address', '')}>"
                    )

            email_content_parts.append(

                f"Subject: {subject}\n"
                f"Content:\n{body[:1500]}\n"
            )

            email_content=full_email_content,
```

```python
followup_tools.py
=================
MCP tools for tracking email follow-ups — scanning for unanswered

async def _needs_followup(message: dict) -> bool:
    """
    Determine if an email needs a follow-up by checking:
    1. Email is older than FOLLOWUP_THRESHOLD_HOURS
    2. No reply exists in the same conversation thread from the user
    """
    # Check age
    received_raw = message.get("receivedDateTime", "")
    if not received_raw:
        return False

    try:
        from dateutil import parser as dp
        received_dt = dp.parse(received_raw)
        # Make timezone-aware for comparison
        if received_dt.tzinfo is None:
            received_dt = received_dt.replace(tzinfo=timezone.utc)
        age_hours = (
            datetime.now(timezone.utc) - received_dt
        ).total_seconds() / 3600

        if age_hours < FOLLOWUP_THRESHOLD_HOURS:
            return False

    except Exception:
        return False

    return True

    """
    logger.info(
        f"Tool called: scan_followups "

    )

    try:
        safe_count = min(max(count, 1), 50)
        raw_messages = await fetch_recent_messages(top=safe_count)

        if not raw_messages:
            return {
                "followups": [],
                "count": 0,
                "display_table": "📭 No emails found to scan.",
                "instruction": "Inform the user their inbox appears empty.",
            }

        # Filter emails needing follow-up
        followups = []
        now = datetime.now(timezone.utc)

        for msg in raw_messages:
            received_raw = msg.get("receivedDateTime", "")
            if not received_raw:
                continue

            try:
                from dateutil import parser as dp
                received_dt = dp.parse(received_raw)
                if received_dt.tzinfo is None:
                    received_dt = received_dt.replace(tzinfo=timezone.utc)
                age_hours = (now - received_dt).total_seconds() / 3600
            except Exception:
                continue

            # Only include emails older than threshold
            if age_hours < older_than_hours:
                continue

            # Calculate age display
            if age_hours < 48:
                age_label = f"{int(age_hours)}h ago"
            else:
                age_label = f"{int(age_hours / 24)}d ago"

            followups.append({
                "id": msg.get("id"),
                "subject": msg.get("subject") or "(no subject)",
                "sender": (

                    .get("emailAddress", {})
                    .get("name") or "—"

                    msg.get("from", {})
                    .get("emailAddress", {})
                    .get("address") or "—"
                ),
                "received_date": received_raw,
                "age_label": age_label,
                "age_hours": round(age_hours, 1),

                "followups": [],
                "count": 0,
                "display_table": (
                    f"✅ No emails older than {older_than_hours} hours "
                    f"found needing follow-up."
                ),
                "instruction": (
                    "Inform the user they have no pending follow-ups in "
                    f"the last {older_than_hours} hours."
                ),
            }

        # Sort by age — oldest first (highest priority)
        followups.sort(key=lambda x: x["age_hours"], reverse=True)

        # Build display table
        table_lines = [

            "|---|-----|------|---------|",

                f"| {i} | ⏰ {f['age_label']} | {f['sender']} | {subject_col} |"
            )

        return {
            "followups": followups,
            "count": len(followups),
            "scanned": len(raw_messages),
            "threshold_hours": older_than_hours,
            "display_table": "\n".join(table_lines),
            "instruction": (

                "or offer to flag any of them for follow-up."
```

```python
@mcp.tool
async def list_tasks(
    source: str = "todo",
    count: int = 20,
) -> dict:
    """
    List tasks from Microsoft To-Do or Microsoft Planner.

    Use this tool when the user asks things like:
    - "Show my To-Do tasks"
    - "What tasks do I have in Planner?"
    - "List my pending tasks"
    - "What's on my task list?"

    Args:
        source (str): Which service to fetch tasks from:
                       - "todo"    → Microsoft To-Do (personal)
                       - "planner" → Microsoft Planner (team)
                       - "both"    → Both services
                       Default: "todo"
        count (int): Maximum tasks to return. Default 20.

    Returns:
        dict with task list and display table.
    """
    logger.info(f"Tool called: list_tasks (source={source}, count={count})")

    try:
        safe_count = min(max(count, 1), 50)
        source_lower = source.lower().strip()

        todo_tasks = []
        planner_tasks = []

        if source_lower in ("todo", "both"):
            todo_tasks = await get_todo_tasks(top=safe_count)

        if source_lower in ("planner", "both"):
            planner_tasks = await get_planner_tasks(top=safe_count)

        # Build display tables
        output_parts = []

        if todo_tasks:
            todo_table = [
                "**📋 Microsoft To-Do Tasks**\n",
                "| # | Task | Due Date | Status |",
                "|---|------|----------|--------|",
            ]
            for i, t in enumerate(todo_tasks, 1):
                status_icon = "🔄" if t["status"] == "inProgress" else "⬜"
                todo_table.append(
                    f"| {i} | {t['title'][:60]} | {t['due_date']} | {status_icon} |"
                )
            output_parts.append("\n".join(todo_table))

        if planner_tasks:
            planner_table = [
                "**📌 Microsoft Planner Tasks**\n",
                "| # | Task | Due Date | Progress |",
                "|---|------|----------|----------|",
            ]
            for i, t in enumerate(planner_tasks, 1):
                progress = f"{t['percent_complete']}%"
                planner_table.append(
                    f"| {i} | {t['title'][:60]} | {t['due_date']} | {progress} |"
                )
            output_parts.append("\n".join(planner_table))

        if not todo_tasks and not planner_tasks:
            display = "✅ No pending tasks found."
        else:
            display = "\n\n".join(output_parts)

        return {
            "todo_tasks": todo_tasks,
            "planner_tasks": planner_tasks,
            "total_count": len(todo_tasks) + len(planner_tasks),
            "display_table": display,
            "instruction": (
                "Display 'display_table' as markdown. "
                "Offer to add new tasks or mark any as complete."
            ),
        }

    except Exception as exc:
        return format_tool_error(exc, tool_name="list_tasks", logger=logger)
```
