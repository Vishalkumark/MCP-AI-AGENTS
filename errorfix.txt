async def call_tool(client: httpx.AsyncClient, tool_name: str, arguments: dict = {}, session_id: str = "") -> dict:
    """
    Send one MCP tool call and return the result.
    FastMCP streamable-http returns SSE (Server-Sent Events) format,
    so we parse the event stream to extract the JSON result.
    """

    headers = {**HEADERS}
    if session_id:
        headers["mcp-session-id"] = session_id

    payload = {
        "jsonrpc": "2.0",
        "id": 1,
        "method": "tools/call",
        "params": {
            "name": tool_name,
            "arguments": arguments,
        }
    }

    response = await client.post(MCP_URL, headers=headers, json=payload)
    response.raise_for_status()

    content_type = response.headers.get("content-type", "")

    # ── Case 1: Plain JSON response ──────────────────────────────
    if "application/json" in content_type:
        data = response.json()
        raw = data.get("result", {}).get("content", [{}])[0].get("text", "{}")
        try:
            return json.loads(raw)
        except Exception:
            return {"raw": raw}

    # ── Case 2: SSE (Server-Sent Events) response ────────────────
    # FastMCP streamable-http sends results as SSE lines like:
    #   data: {"jsonrpc":"2.0","id":1,"result":{...}}
    # We read all lines and find the one that contains the result.
    text = response.text
    for line in text.splitlines():
        line = line.strip()
        if line.startswith("data:"):
            json_str = line[len("data:"):].strip()
            if not json_str:
                continue
            try:
                data = json.loads(json_str)
                # Look for the result payload inside the SSE data
                result = data.get("result", {})
                content = result.get("content", [{}])
                raw = content[0].get("text", "{}") if content else "{}"
                try:
                    return json.loads(raw)
                except Exception:
                    return {"raw": raw}
            except json.JSONDecodeError:
                continue

    # ── Case 3: Nothing parseable found ─────────────────────────
    return {"error": True, "message": f"Unparseable response: {text[:200]}"}

        #   data: {"jsonrpc":"2.0","result":{...}}
        #
        # Lines arrive separately so we collect ALL data: lines
        # and parse the first one that contains a valid result.
        lines = response.text.splitlines()
        for line in lines:
            line = line.strip()
            if not line.startswith("data:"):
                continue
            json_str = line[len("data:"):].strip()
            if not json_str:
                continue
            try:
                data = json.loads(json_str)
                # Skip error responses
                if "error" in data:
                    error_msg = data["error"].get("message", "Unknown error")
                    return json.dumps({"error": True, "message": error_msg})
                content = data.get("result", {}).get("content", [{}])
                if content:
                    return content[0].get("text", "{}")
            except json.JSONDecodeError:
                continue

    return "{}"

    # Step 3: Decode attachment content into raw bytes.
    # The Graph SDK can return content_bytes in three different forms
    # depending on the file type and SDK version:
    #   - bytes      : already raw bytes, use directly
    #   - ASCII str  : base64-encoded string, decode it
    #   - Unicode str: plain text content returned directly (e.g. .txt files)
    content_bytes_b64 = attachment.content_bytes

    if isinstance(content_bytes_b64, bytes):
        # Already raw bytes — use directly
        file_bytes = content_bytes_b64

    elif isinstance(content_bytes_b64, str):
        # Check if it's pure ASCII — if so treat as base64
        try:
            content_bytes_b64.encode("ascii")
            # Pure ASCII — treat as base64, fix padding if needed
            padding_needed = len(content_bytes_b64) % 4
            if padding_needed:
                content_bytes_b64 += "=" * (4 - padding_needed)
            file_bytes = base64.b64decode(content_bytes_b64)

        except UnicodeEncodeError:
            # Contains non-ASCII — this IS the plain text content
            # returned directly by the SDK (common for .txt files)
            file_bytes = content_bytes_b64.encode("utf-8")

    else:
        raise ValueError(
            f"Unexpected content_bytes type from Graph SDK: "
            f"{type(content_bytes_b64)}. Cannot decode attachment."
        )

### Workflow for a single file change

```bash
cd /opt/FiGPT_OutlookMCP

# See what changed
git status

# Stage the specific file changed
git add tools/email_tools.py

# Commit with a clear message
git commit -m "feat: add search_emails_advanced with date range filter"

# Push to Azure DevOps
git push
```

### Workflow for multiple file changes

```bash
# Stage everything changed
git add .

# Or stage specific files
git add tools/email_tools.py graph/mail_client.py

# Commit
git commit -m "feat: add pagination and export tools (Points 4 and 5)"

# Push
git push
```

### Check commit history anytime

```bash
git log --oneline
```

Use this format for clean, readable history in Azure DevOps:

|Prefix     |When to use        |Example                                                   |
|-----------|-------------------|----------------------------------------------------------|
|`feat:`    |New tool or feature|`feat: add list_emails_paged tool`                        |
|`fix:`     |Bug fix            |`fix: base64 padding error in attachment download`        |
|`auth:`    |Auth related       |`auth: add X-Fichtner-Authorization header support`       |
|`config:`  |Config/env change  |`config: update REQUESTY_PRIMARY_MODEL in .env`           |
|`docs:`    |Documentation      |`docs: add Phase 1 execution guide`                       |
|`test:`    |Test changes       |`test: fix pdf parser test for PyMuPDF 1.24+`             |
|`refactor:`|Code cleanup       |`refactor: remove internal LLM calls from summarise tools`|
|`deploy:`  |Deployment change  |`deploy: add FASTMCP_HOST env var for nginx 421 fix`      |

## Part H — Suggested Branch Strategy (Simple, Solo Developer)

Since you’re the only developer right now:

```
main (or master)
  └── your commits go directly here
```

When Phase 2 begins or another developer joins:

```
main
  ├── feature/phase2-calendar-write
  ├── feature/phase2-sent-items
  └── hotfix/attachment-read-error
```

Create a branch:

```bash
git checkout -b feature/phase2-calendar-write
```

Merge back to main when done:

```bash
git checkout main
git merge feature/phase2-calendar-write
git push
```

## Part I — Quick Reference Card

```bash
# Daily workflow
git status                          # what changed?
git add .                           # stage all changes
git add <file>                      # stage one file
git commit -m "type: description"   # commit
git push                            # push to Azure DevOps

# Check history
git log --oneline                   # short commit history
git log --oneline -10               # last 10 commits

# Undo staged file (before commit)
git restore --staged <file>

# Undo last commit (keep changes)
git reset --soft HEAD~1

# See what changed in a file
git diff <file>
```

## What NOT to commit — ever

|File/Folder           |Reason                                                            |
|----------------------|------------------------------------------------------------------|
|`.env`                |Contains Azure secrets and API keys                               |
|`bin/token.json`      |Contains your personal Microsoft OAuth token                      |
|`.venv/`              |Python packages — recreated with `pip install -r requirements.txt`|
|`__pycache__/`        |Python bytecode — auto-generated                                  |
|`temp_attachments/*`  |Temporary downloaded files                                        |
|`bin/agent_output.txt`|Local test output                                                 |

All of these are already in `.gitignore`. The `git status` check in
Step D.3 is your safety net before every push.
